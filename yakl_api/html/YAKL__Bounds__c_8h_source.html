9  // LBnd: Loop Bound -- Describes the bounds of one loop
10  // C defaults to lower bound of 0
12  class LBnd {
13  public:
14  int static constexpr default_lbound = 0;
15  int l, u, s;
16  YAKL_INLINE LBnd() {
17  this->l = -1;
18  this->u = -1;
19  this->s = -1;
20  }
21  // Lower bound of zero, stride of one
22  YAKL_INLINE LBnd(index_t u) {
23  this->l = 0;
24  this->u = u-1;
25  this->s = 1;
26  }
27  // Lower bound of zero, stride of one
28  YAKL_INLINE LBnd(int u) {
29  this->l = 0;
30  this->u = u-1;
31  this->s = 1;
32  }
33  // Lower and upper bounds specified, stride of one
34  YAKL_INLINE LBnd(int l, int u) {
35  this->l = l;
36  this->u = u;
37  this->s = 1;
38  if (u < l) yakl_throw("ERROR: cannot specify an upper bound < lower bound");
39  }
40  // Lower bound, upper bound, and stride all specified
41  YAKL_INLINE LBnd(int l, int u, int s) {
42  this->l = l;
43  this->u = u;
44  this->s = s;
45  if (s < 1) yakl_throw("ERROR: negative strides not yet supported.");
46  }
47  YAKL_INLINE index_t to_scalar() const {
48  return (index_t) u+1;
49  }
50  YAKL_INLINE bool valid() const { return this->s > 0; }
51  };
