685  template <int N> YAKL_INLINE Array<T,N,myMem,styleC> slice( Dims const &dims ) const {
686  #ifdef YAKL_DEBUG
687  if (rank != dims.size()) {
688  #if YAKL_CURRENTLY_ON_HOST()
689  std::cerr << "For Array named " << this->myname << ": ";
690  #endif
691  yakl_throw("ERROR: slice rank must be equal to dims.size()");
692  }
693  for (int i = rank-1-N; i >= 0; i--) {
694  if (dims.data[i] >= this->dimension[i]) {
695  #if YAKL_CURRENTLY_ON_HOST()
696  std::cerr << "For Array named " << this->myname << ": ";
697  #endif
698  yakl_throw("ERROR: One of the slicing dimension dimensions is out of bounds");
699  }
700  }
701  if (! this->initialized()) {
702  #if YAKL_CURRENTLY_ON_HOST()
703  std::cerr << "For Array named " << this->myname << ": ";
704  #endif
705  yakl_throw("ERROR: calling slice() on an Array that hasn't been allocated");
706  }
707  #endif
708  Array<T,N,myMem,styleC> ret;
709  index_t offset = 1;
710  for (int i = rank-1; i > rank-1-N; i--) {
711  ret.dimension[i-(rank-N)] = this->dimension[i];
712  offset *= this->dimension[i];
713  }
714  index_t retOff = 0;
715  for (int i = rank-1-N; i >= 0; i--) {
716  retOff += dims.data[i]*offset;
717  offset *= this->dimension[i];
718  }
719  ret.myData = &(this->myData[retOff]);
720  #if YAKL_CURRENTLY_ON_HOST()
721  yakl_mtx_lock();
722  ret.refCount = this->refCount;
723  if (this->refCount != nullptr) {
724  (*(this->refCount))++;
725  }
726  yakl_mtx_unlock();
727  #endif
728  return ret;
729  }

733  static_assert( rank == 1 , "ERROR: Calling slice() with 1 index on a non-rank-1 array" );
734  static_assert( N <= rank , "ERROR: Calling slice() with more dimenions than this array's rank" );
