582  template <int N> YAKL_INLINE Array<T,N,myMem,styleFortran> slice( Dims const &dims ) const {
584  #ifdef YAKL_DEBUG
585  if (rank != dims.size()) {
586  #if YAKL_CURRENTLY_ON_HOST()
587  std::cerr << "For Array named " << this->myname << ": ";
588  #endif
589  yakl_throw("ERROR: rank must be equal to dims.size()");
590  }
591  for (int i=N; i<rank; i++) {
592  if (dims.data[i] < this->lbounds[i] || dims.data[i] >= this->lbounds[i]+this->dimension[i] ) {
593  #if YAKL_CURRENTLY_ON_HOST()
594  std::cerr << "For Array named " << this->myname << ": ";
595  #endif
596  yakl_throw("ERROR: One of the slicing dimension dimensions is out of bounds");
597  }
598  }
599  if (! this->initialized()) {
600  #if YAKL_CURRENTLY_ON_HOST()
601  std::cerr << "For Array named " << this->myname << ": ";
602  #endif
603  yakl_throw("ERROR: calling slice() on an Array that hasn't been allocated");
604  }
605  #endif
606  Array<T,N,myMem,styleFortran> ret;
607  index_t offset = 1;
608  for (int i=0; i<N; i++) {
609  ret.dimension[i] = this->dimension[i];
610  ret.lbounds [i] = this->lbounds [i];
611  offset *= this->dimension[i];
612  }
613  index_t retOff = 0;
614  for (int i=N; i<rank; i++) {
615  retOff += (dims.data[i]-this->lbounds[i])*offset;
616  offset *= this->dimension[i];
617  }
618  ret.myData = &(this->myData[retOff]);
619  #if YAKL_CURRENTLY_ON_HOST()
620  yakl_mtx_lock();
621  ret.refCount = this->refCount;
622  if (this->refCount != nullptr) {
623  (*(this->refCount))++;
624  }
625  yakl_mtx_unlock();
626  #endif
627  return ret;
628  }

703  #endif
