654  template <int N> YAKL_INLINE Array<T,N,myMem,styleFortran> slice( Dims const &dims ) const {
655  #ifdef YAKL_DEBUG
656  if (rank != dims.size()) {
657  #if YAKL_CURRENTLY_ON_HOST()
658  std::cerr << "For Array named " << this->myname << ": ";
659  #endif
660  yakl_throw("ERROR: rank must be equal to dims.size()");
661  }
662  for (int i=N; i<rank; i++) {
663  if (dims.data[i] < this->lbounds[i] || dims.data[i] >= this->lbounds[i]+this->dimension[i] ) {
664  #if YAKL_CURRENTLY_ON_HOST()
665  std::cerr << "For Array named " << this->myname << ": ";
666  #endif
667  yakl_throw("ERROR: One of the slicing dimension dimensions is out of bounds");
668  }
669  }
670  if (! this->initialized()) {
671  #if YAKL_CURRENTLY_ON_HOST()
672  std::cerr << "For Array named " << this->myname << ": ";
673  #endif
674  yakl_throw("ERROR: calling slice() on an Array that hasn't been allocated");
675  }
676  #endif
677  Array<T,N,myMem,styleFortran> ret;
678  index_t offset = 1;
679  for (int i=0; i<N; i++) {
680  ret.dimension[i] = this->dimension[i];
681  ret.lbounds [i] = this->lbounds [i];
682  offset *= this->dimension[i];
683  }
684  index_t retOff = 0;
685  for (int i=N; i<rank; i++) {
686  retOff += (dims.data[i]-this->lbounds[i])*offset;
687  offset *= this->dimension[i];
688  }
689  ret.myData = &(this->myData[retOff]);
690  #if YAKL_CURRENTLY_ON_HOST()
691  yakl_mtx_lock();
692  ret.refCount = this->refCount;
693  if (this->refCount != nullptr) {
694  (*(this->refCount))++;
695  }
696  yakl_mtx_unlock();
697  #endif
698  return ret;
699  }

790  if (! this->initialized()) { yakl_throw("ERROR: Trying to reshape an Array that hasn't been initialized"); }

795  #endif

797  for (int i=0; i < N; i++) {
798  ret.dimension[i] = bnds.u[i] - bnds.l[i] + 1; ret.lbounds [i] = bnds.l[i];
799  }

802  #endif
803  ret.myData = this->myData;
