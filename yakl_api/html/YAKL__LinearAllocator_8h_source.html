173  // Allocate the requested number of bytes if there is room for it.
174  // If there isn't room or bytes == 0, then nullptr is returned.
175  // Otherwise, the correct pointer is returned
176  void * allocate(size_t bytes, char const * label="") {
177  if (bytes == 0) {
178  return nullptr;
179  }
180  size_t blocksReq = (bytes-1)/blockSize + 1; // Number of blocks needed for this allocation
181  // If there are no allocations, then place this allocaiton at the beginning.
182  if (allocs.empty()) {
183  if (nBlocks >= blocksReq) {
184  allocs.push_back( AllocNode( (size_t) 0 , blocksReq , label ) );
185  return pool;
186  }
187  } else {
188  // Look for room before the first allocation
189  if ( allocs.front().start >= blocksReq ) {
190  allocs.insert( allocs.begin() , AllocNode( 0 , blocksReq , label ) );
191  return getPtr(allocs[0].start);
192  }
193 
194  // Loop through the allocations and look for free space between this and the next
195  for (int i=0; i < allocs.size()-1; i++) {
196  if ( allocs[i+1].start - (allocs[i].start + allocs[i].length) >= blocksReq ) {
197  allocs.insert( allocs.begin()+i+1 , AllocNode( allocs[i].start+allocs[i].length , blocksReq , label ) );
198  return getPtr(allocs[i+1].start);
199  }
200  }
201 
202  // Look for room after the last allocation
203  if ( nBlocks - (allocs.back().start + allocs.back().length) >= blocksReq ) {
204  allocs.push_back( AllocNode( allocs.back().start + allocs.back().length , blocksReq , label ) );
205  return getPtr(allocs.back().start);
206  }
207  }
208 
209  // Return nullptr if there was no room for the allocation
210  // If the caller used "iGotRoom", then this should never actually happen
211  return nullptr;
212  };

217  for (int i=allocs.size()-1; i >= 0; i--) {
