392  dims[i] = tmp;

403  auto varDims = var.getDims();
404  if (varDims.size() != rank) { yakl_throw("Existing variable's rank != array's rank"); }

529  } else if (myStyle == styleFortran) {

539  createArr = true;
540  }
541  }
542  }

544  } else { yakl_throw("Variable does not exist"); }
545

547  auto arrHost = arr.createHostCopy();
