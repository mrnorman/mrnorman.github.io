54  void start(std::string label) {
55  if (label.empty()) die("ERROR: calling start() with an empty label");

57  auto thread_id = std::this_thread::get_id();
58  auto label_hash = hasher( label );
59  auto thread_index = get_or_create_thread_index( thread_id );
60  auto timer_index = get_or_create_timer_index( thread_index , label , label_hash );
61  auto &timer = threads[thread_index].timers[timer_index];
62  auto &active_stack = threads[thread_index].active_stack;
63  // If the active stack isn't empty, then register my timer as a child of the stack top timer
64  if ( ! active_stack.empty() ) {
65  int parent_timer_index = active_stack.back().timer_index;
66  auto &parent_timer = threads[thread_index].timers[parent_timer_index];
67  bool i_am_already_a_child = false;
68  for ( auto &child_hash : parent_timer.child_hashes ) {
69  if ( child_hash == label_hash ) { i_am_already_a_child = true; break; }
70  }
71  if ( ! i_am_already_a_child ) parent_timer.child_hashes.push_back( label_hash );
72  if (timer.parent_index == parent_index_just_created) timer.parent_index = parent_timer_index;
73  if (timer.parent_index != parent_timer_index) timer.multiple_parents = true;
74  } else {
75  if (timer.parent_index == parent_index_just_created) timer.parent_index = parent_index_main;
76  if (timer.parent_index != parent_index_main) timer.multiple_parents = true;
77  }
78  // Push this timer onto the active stack
79  active_stack.push_back( { label_hash , timer_index } );
80  // Register the time point to start the timer
81  timer.hits++;
82  timer.previous_time_point = Clock::now();

86  void stop(std::string label) {
87  TimePoint now = Clock::now();
88  if (label.empty()) die("ERROR: calling stop() with an empty label");
89  auto thread_id = std::this_thread::get_id();
90  auto label_hash = hasher( label );
91  auto thread_index = get_or_create_thread_index( thread_id );
92  auto &active_stack = threads[thread_index].active_stack;
93  // Make sure the timers are properly nested to guarantee the active stack's back timer_index is appropriate
94  if ( label_hash != active_stack.back().label_hash ) die("ERROR: timers must be perfectly nested");
95  // Instead of searching for the timer index, use the one from the back of the active stack
96  auto &timer = threads[thread_index].timers[active_stack.back().timer_index];
97  // Set the timer stats
98  Duration duration = now - timer.previous_time_point;
99  timer.max_duration = max( timer.max_duration , duration );
100  timer.min_duration = min( timer.min_duration , duration );
101  timer.accumulated_duration += duration;
102  // Remove this timer from the active stack
103  active_stack.pop_back();
104  }
