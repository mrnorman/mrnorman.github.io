12 YAKL_DEVICE_INLINE void callFunctor(F const &f , Bounds<N,simple> const &bnd , int const i ) {
13  int ind[N];
14  bnd.unpackIndices( i , ind );
15  if constexpr (N == 1) f(ind[0]);
16  if constexpr (N == 2) f(ind[0],ind[1]);
17  if constexpr (N == 3) f(ind[0],ind[1],ind[2]);
18  if constexpr (N == 4) f(ind[0],ind[1],ind[2],ind[3]);
19  if constexpr (N == 5) f(ind[0],ind[1],ind[2],ind[3],ind[4]);

21  if constexpr (N == 7) f(ind[0],ind[1],ind[2],ind[3],ind[4],ind[5],ind[6]);
22  if constexpr (N == 8) f(ind[0],ind[1],ind[2],ind[3],ind[4],ind[5],ind[6],ind[7]);

46  // CUDA has a limit on the parameter space for a kernel launch. When it's exceeded, then the kernel
47  // needs to be launched by reference from device memory (i.e., "cudaKernelRef").
48  // otherwise, it needs to be launched by value (i.e., "cudaKernelVal")
49  // A kernel launch will get the global ID and then use the callFunctor intermediate to tranform that
50  // ID into a set of indices for multiple loops and then call the functor with those indices
51  // The __launch_bounds__ matches the number of threads used to launch the kernels so that the compiler
52  // does not compile for more threads per block than the user plans to use. This is for optimal register
53  // usage and reduced register spilling.

55  void cudaKernelVal( Bounds<N,simple> bounds , F f , LaunchConfig<VecLen,B4B> config ) {
56  size_t i = blockIdx.x*blockDim.x + threadIdx.x;
57  if (i < bounds.nIter) {
58  callFunctor( f , bounds , i );
59  }
60  }
