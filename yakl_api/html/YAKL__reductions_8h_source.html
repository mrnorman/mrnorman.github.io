19  // These are parallel reduction classes for host and device
20  // They handle all memory allocation and deallocation internally to reduce the chance of memory errors
21  // setup and constructor take an integer of the number of items needing to be reduced
22  // operator() will perform a reduction on the device and return the result as a scalar value
23  // after copying to the host
24  // deviceReduce will perform a reduciton on the device and leave hte result on the device as a pointer
25  // Array is passed to the device as a pointer, so it's not completely safe is the user makes an error
26  // where the repested nItems differs from the actual number of items in the array.

29  template <class T, int RED> class ParallelReduction<T,memHost,RED> {
30  int nItems; // Number of items in the array that will be reduced
31  public:
32  ParallelReduction() {}
33  ParallelReduction(int const nItems) { this->nItems = nItems; }
34  ~ParallelReduction() {}
35  void setup(int nItems) { this->nItems = nItems; }
36  T operator() (T *data) {
37  T rslt = data[0];
38  if constexpr (RED == YAKL_REDUCTION_MIN) {
39  for (int i=1; i<nItems; i++) { rslt = data[i] < rslt ? data[i] : rslt; }
40  } else if constexpr (RED == YAKL_REDUCTION_MAX) {
41  for (int i=1; i<nItems; i++) { rslt = data[i] > rslt ? data[i] : rslt; }
42  } else if constexpr (RED == YAKL_REDUCTION_SUM) {
43  for (int i=1; i<nItems; i++) { rslt += data[i]; }
44  } else if constexpr (RED == YAKL_REDUCTION_PROD) {
45  for (int i=1; i<nItems; i++) { rslt *= data[i]; }
46  }
47  return rslt;
48  }
49  };

79  }
80  tmp = alloc_device(nTmp,"Parallel Reduction Temporary"); // Allocate temporary storage

84  #endif

99  hipcub::DeviceReduce::Min(tmp, nTmp, data , rsltP , nItems , 0 ); // Compute the reduction

102  } else if constexpr (RED == YAKL_REDUCTION_SUM) {

104  } else if constexpr (RED == YAKL_REDUCTION_PROD) {
105  hipcub::DeviceReduce::Reduce(tmp, nTmp, data, rsltP, nItems, YAKL_LAMBDA (T a,T b)->T {return a*b;} , (T) 1, 0 );
106  }
107  memcpy_device_to_host(&rslt , rsltP , 1 );
108  check_last_error();

227  } else if constexpr (RED == YAKL_REDUCTION_SUM) {
228  cgh.parallel_for(sycl::range<1>(nItems),
229  sycl::reduction(rsltP, std::plus<>(),
230  sycl::property::reduction::initialize_to_identity{}),

234  sycl::reduction(rsltP, std::multiplies<>(),
