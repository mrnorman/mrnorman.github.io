173  void * allocate(size_t bytes, char const * label="") {
174  if (bytes == 0) return nullptr;
175  if (! waiting_events.empty()) free_completed_waiting_entries();
176  // Loop through the pools and see if there's room. If so, allocate in one of them
177  bool room_found = false; // Whether room exists for the allocation
178  bool linear_bug = false; // Whether there's an apparent bug in the LinearAllocator allocate() function
179  void *ptr; // Allocated pointer
180  // Protect against multiple threads trying to allocate at the same time
181  mtx1.lock();
182  {
183  // Start at the first pool, see if it has room.
184  // If so, allocate in that pool and break. If not, try the next pool.
185  for (auto it = pools.begin() ; it != pools.end() ; it++) {
186  if (it->iGotRoom(bytes)) {
187  ptr = it->allocate(bytes,label);
188  room_found = true;
189  if (ptr == nullptr) linear_bug = true;
190  break;
191  }
192  }
193  // If you've gone through all of the existing pools, and room hasn't been found, then it's time to add a new pool
194  if (!room_found) {
195  if (bytes > growSize) {
196  std::cerr << "ERROR: For the pool allocator labeled \"" << pool_name << "\":" << std::endl;
197  std::cerr << "ERROR: Trying to allocate " << bytes << " bytes (" << bytes/1024./1024./1024. << " GB), "
198  << "but the current pool is too small, and growSize is only "
199  << growSize << " bytes (" << growSize/1024./1024./1024. << " GB). \nThus, the allocation will never fit in pool memory.\n";
200  std::cerr << "This can happen for a number of reasons. \nCheck the size of the variable being allocated in the "
201  << "line above and see if it's what you expected. \nIf it's absurdly large, then you might have tried "
202  << "to pass in a negative value for the size, or the size got corrupted somehow. \nNOTE: If you compiled "
203  << "for the wrong GPU artchitecture, it sometimes shows up here as well. \nIf the size of the variable "
204  << "is realistic, then you should increase the initial pool size and probably the grow size as "
205  << "well. \nWhen individual variables consume sizable percentages of a pool, memory gets fragmented, and "
206  << "the pool space isn't used efficiently. \nLarger pools will improve that. "
207  << "\nIn the extreme, you could create "
208  << "an initial pool that consumes most of the avialable device memory. \nIf that still doesn't work, then "
209  << "it sounds like you're choosing a problem size that's too large for the number of compute "
210  << "nodes you're using.\n";
211  std::cerr << error_message_cannot_grow << std::endl;
212  printAllocsLeft();
213  die();
214  }
215  pools.push_back( LinearAllocator( growSize , blockSize , mymalloc , myfree , myzero ,
216  pool_name , error_message_out_of_memory) );
217  ptr = pools.back().allocate(bytes,label);
218  }
219  bytes_currently_allocated += ( (bytes-1)/blockSize+1 )*blockSize;
220  high_water_mark = std::max( high_water_mark , bytes_currently_allocated );
221  }
222  mtx1.unlock();
223  if (linear_bug) {
224  std::cerr << "ERROR: For the pool allocator labeled \"" << pool_name << "\":" << std::endl;
225  die("ERROR: It looks like you've found a bug in LinearAllocator. Please report this at github.com/mrnorman/YAKL");
226  }
227  if (ptr != nullptr) {
228  return ptr;
229  } else {
230  std::cerr << "ERROR: For the pool allocator labeled \"" << pool_name << "\":" << std::endl;
231  std::cerr << "Unable to allocate pointer. It looks like you might have run out of memory.";
232  die( error_message_out_of_memory );
233  }
234  return nullptr;
235  };

282  void free_completed_waiting_entries() {
283  mtx2.lock();
284  // Loop through waiting events. Check if it's completed
285  for (int ievent=0; ievent < waiting_events.size(); ievent++) {
286  auto waiting_event = waiting_events[ievent];
287  // If this event has completed, erase all instances of this event from all waiting entry event lists
288  if (waiting_event.completed()) {
289  auto &completed_event = waiting_event;
290  // Loop through waiting entries and erase completed event from each entry's waiting event list
291  for (int ientry=0; ientry < waiting_entries.size(); ientry++) {
292  auto waiting_entry = waiting_entries[ientry];
293  // Search through this waiting entry's waiting event list, and erase any occurrences of this completed event
294  for (int i=0; i < waiting_entry.events.size(); i++) {
295  if (waiting_entry.events[i] == completed_event) waiting_entry.events.erase(waiting_entry.events.begin()+i);
296  break;
297  }
298  // if this waiting entry's waiting event list is empty, free its pointer & erase it from waiting entry list
299  if (waiting_entry.events.empty()) {
300  this->free( waiting_entry.ptr , waiting_entry.label.c_str() );
301  waiting_entries.erase( waiting_entries.begin()+ientry );
302  }
303  }
304  // Finally, erase this completed event from the waiting events list
305  waiting_events.erase(waiting_events.begin()+ievent);
306  }
307  }
308  mtx2.unlock();
309  }
