139  void * allocate(size_t bytes, char const * label="") {
140  if (bytes == 0) return nullptr;
141  // Loop through the pools and see if there's room. If so, allocate in one of them
142  bool room_found = false; // Whether room exists for the allocation
143  bool linear_bug = false; // Whether there's an apparent bug in the LinearAllocator allocate() function
144  void *ptr; // Allocated pointer
145  // Protect against multiple threads trying to allocate at the same time
146  mtx.lock();
147  {
148  // Start at the first pool, see if it has room.
149  // If so, allocate in that pool and break. If not, try the next pool.
150  for (auto it = pools.begin() ; it != pools.end() ; it++) {
151  if (it->iGotRoom(bytes)) {
152  ptr = it->allocate(bytes,label);
153  room_found = true;
154  if (ptr == nullptr) linear_bug = true;
155  break;
156  }
157  }
158  // If you've gone through all of the existing pools, and room hasn't been found, then it's time to add a new pool
159  if (!room_found) {
160  if (bytes > growSize) {
161  std::cerr << "ERROR: For the pool allocator labeled \"" << pool_name << "\":" << std::endl;
162  std::cerr << "ERROR: Trying to allocate " << bytes << " bytes (" << bytes/1024./1024./1024. << " GB), "
163  << "but the current pool is too small, and growSize is only "
164  << growSize << " bytes (" << growSize/1024./1024./1024. << " GB). \nThus, the allocation will never fit in pool memory.\n";
165  std::cerr << "This can happen for a number of reasons. \nCheck the size of the variable being allocated in the "
166  << "line above and see if it's what you expected. \nIf it's absurdly large, then you might have tried "
167  << "to pass in a negative value for the size, or the size got corrupted somehow. \nNOTE: If you compiled "
168  << "for the wrong GPU artchitecture, it sometimes shows up here as well. \nIf the size of the variable "
