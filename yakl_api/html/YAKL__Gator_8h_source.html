81  // Allocate memory with the specified number of bytes and the specified label
82  void * allocate(size_t bytes, char const * label="") {
83  if (bytes == 0) return nullptr;
84  // Loop through the pools and see if there's room. If so, allocate in one of them
85  bool room_found = false; // Whether room exists for the allocation
86  bool linear_bug = false; // Whether there's an apparent bug in the LinearAllocator allocate() function
87  void *ptr; // Allocated pointer
88  // Protect against multiple threads trying to allocate at the same time
89  mtx.lock();
90  {
91  // Start at the first pool, see if it has room.
92  // If so, allocate in that pool and break. If not, try the next pool.
93  for (auto it = pools.begin() ; it != pools.end() ; it++) {
94  if (it->iGotRoom(bytes)) {
95  ptr = it->allocate(bytes,label);
96  room_found = true;
97  if (ptr == nullptr) linear_bug = true;
98  break;
99  }
100  }
101  // If you've gone through all of the existing pools, and room hasn't been found, then it's time to add a new pool
102  if (!room_found) {
103  if (bytes > growSize) {
104  std::cerr << "ERROR: For the pool allocator labeled \"" << pool_name << "\":" << std::endl;
105  std::cerr << "ERROR: Trying to allocate " << bytes << " bytes (" << bytes/1024./1024./1024. << " GB), "
106  << "but the current pool is too small, and growSize is only "
107  << growSize << " bytes (" << growSize/1024./1024./1024. << " GB). \nThus, the allocation will never fit in pool memory.\n";
108  std::cerr << "This can happen for a number of reasons. \nCheck the size of the variable being allocated in the "
109  << "line above and see if it's what you expected. \nIf it's absurdly large, then you might have tried "
110  << "to pass in a negative value for the size, or the size got corrupted somehow. \nNOTE: If you compiled "
111  << "for the wrong GPU artchitecture, it sometimes shows up here as well. \nIf the size of the variable "
112  << "is realistic, then you should increase the initial pool size and probably the grow size as "
113  << "well. \nWhen individual variables consume sizable percentages of a pool, memory gets segmented, and "
114  << "the pool space isn't used efficiently. \nLarger pools will improve that. "
115  << "\nIn the extreme, you could create "
116  << "an initial pool that consumes most of the avialable memory. \nIf that still doesn't work, then "
117  << "it sounds like you're choosing a problem size that's too large for the number of compute "
118  << "nodes you're using.\n";
119  std::cerr << error_message_cannot_grow << std::endl;
120  printAllocsLeft();
121  die();
122  }
123  pools.push_back( LinearAllocator( growSize , blockSize , mymalloc , myfree , myzero ,
124  pool_name , error_message_out_of_memory) );
125  ptr = pools.back().allocate(bytes,label);
126  }
127  }
128  mtx.unlock();
129  if (linear_bug) {
130  std::cerr << "ERROR: For the pool allocator labeled \"" << pool_name << "\":" << std::endl;
131  die("ERROR: It looks like you've found a bug in LinearAllocator. Please report this at github.com/mrnorman/YAKL");
132  }
133  if (ptr != nullptr) {
134  return ptr;
135  } else {
136  std::cerr << "ERROR: For the pool allocator labeled \"" << pool_name << "\":" << std::endl;
137  std::cerr << "Unable to allocate pointer. It looks like you might have run out of memory.";
138  die( error_message_out_of_memory );
139  }
140  return nullptr;
141  };
